const rotateTile = () => {
  let divs = Array.from(document.querySelectorAll('.animate'));
  let test = [];
  divs.forEach(item => {
      if (item.firstElementChild.innerHTML) {
        test.push(item)
      }
    });
  let angle = 0;
  const id = setInterval(() => {
    test.forEach(item => {
      if (angle >= 360) {
        clearInterval(id);
      } 
      else {
        console.log(angle, item)
        angle++;
        item.style.transform = "rotateX(" + angle + "deg)";
      }
    });
  }, 2);
}
